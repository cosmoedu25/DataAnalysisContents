def statistical_significance_test(df_enhanced):

    print("\n2ï¸âƒ£ í†µê³„ì  ìœ ì˜ì„± ê²€ì¦")

    gender_crosstab = pd.crosstab(df_enhanced['Sex'], df_enhanced['Survived'])
